<!-- To write freeze state directly via MCP execute_python, use the JSON knob approach: -->
```python
import json, uuid

# To freeze a group of nodes together:
group_uuid = str(uuid.uuid4())
for node_name in ["Grade1", "Blur1", "Merge1"]:
    node = nuke.toNode(node_name)
    # Ensure the hidden tab and knob exist
    if not node.knob("node_layout_tab"):
        tab = nuke.Tab_Knob("node_layout_tab", "")
        tab.setFlag(nuke.INVISIBLE)
        node.addKnob(tab)
    if not node.knob("node_layout_state"):
        state_knob = nuke.String_Knob("node_layout_state", "")
        state_knob.setFlag(nuke.INVISIBLE)
        node.addKnob(state_knob)
    # Read existing state, set freeze_group, write back
    raw = node.knob("node_layout_state").value()
    state = json.loads(raw) if raw else {"scheme": "normal", "mode": "vertical", "h_scale": 1.0, "v_scale": 1.0, "freeze_group": None}
    state["freeze_group"] = group_uuid
    node.knob("node_layout_state").setValue(json.dumps(state))
```
