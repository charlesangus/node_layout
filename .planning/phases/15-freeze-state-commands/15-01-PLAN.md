```python
        source = source_file.read()
```

```python
class TestFreezeGroupState(unittest.TestCase):

    def test_read_node_state_returns_freeze_group_none_for_old_node_without_key(self):
        """Simulate an old node whose JSON has no freeze_group key."""
        node = FakeNode()

        node = FakeNode()

        state = dict(node_layout_state._DEFAULT_STATE)
        state["freeze_group"] = test_uuid
        node_layout_state.write_node_state(node, state)
        result = node_layout_state.read_node_state(node)
        self.assertEqual(result["freeze_group"], test_uuid)
```

```python
def read_freeze_group(node):
    """Return the freeze group UUID string for *node*, or None if not frozen."""
    return read_node_state(node).get("freeze_group")


def write_freeze_group(node, group_uuid):
    """Set the freeze group UUID on *node*.

    *group_uuid* should be a string (e.g. from ``str(uuid.uuid4())``).
    Pass ``None`` to remove freeze group membership.
    """
    state = read_node_state(node)
    state["freeze_group"] = group_uuid
    write_node_state(node, state)


def clear_freeze_group(node):
    """Remove freeze group membership from *node*.

    Convenience alias for ``write_freeze_group(node, None)``.
    """
    write_freeze_group(node, None)
```

These helpers compose on top of `read_node_state`/`write_node_state` — they never bypass the existing JSON round-trip. No new imports needed in this file.
